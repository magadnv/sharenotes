<script lang="ts">
  import { goto } from '$app/navigation';
  let q = '';
  function onSubmit(e: Event) {
    e.preventDefault();
    const query = q.trim();
    if (!query) return;
    goto(`/search?q=${encodeURIComponent(query)}`);
  }
</script>
<section aria-labelledby="search" class="mx-auto max-w-3xl">
  <form on:submit={onSubmit} class="relative">
    <input
      id="q"
      name="q"
      bind:value={q}
      placeholder="Thema, Fach oder Begriff eingeben …"
      autocomplete="off"
      class="w-full rounded-2xl border border-black/10 bg-white px-5 py-4 text-base outline-none focus:border-black/30"
    />
    <div class="mt-2 text-xs text-black/50">Tippe und drücke Enter.</div>
  </form>

  <!-- Optional: kleine „Beliebte Suchen“-Chips -->
</section>
